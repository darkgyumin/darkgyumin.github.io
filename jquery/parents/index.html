jQuery.parents()
  • 자기자신의 상위의 엘리먼트를 선택한다.
  • selector 입력시 selector와 매칭되는 상위의 엘리먼트를 선택한다.
jQuery.parents([selector]) 작동예제

console.log($('.item-2-1').parents('ul'));
//init(2) [ul.level-2, ul.level-1, prevObject: init(1), context: document]

$('.item-2-1').parents('ul').css('background-color', '#ccc');
//$('.item-2-1').parent('ul')에 의해 li.item-2-1의 상위에 있는 모든 ul엘리멘트가 선택된다.
