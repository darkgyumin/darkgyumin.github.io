jQuery.ajax()
  • 비동기 HTTP 요청을 수행한다.
기본 ajax작동 예제

$.ajax({
	url: 'https://jsonplaceholder.typicode.com/posts',
	data: 'userId=1',
	dataType: "json",
	timeout: '1000',
	type: 'GET',
	success: function(data) {
		console.log(data);
	}
});
Keyword Type Description
url String
data PlainObject or String or Array
dataType PlainObject or String or Array
timeout Number timeout에 설정된 시간이후 요청작업이 처리되지 않으면
success Function
(Anything data, String textStatus, jqXHR jqXHR) 		요청작업이 성공했을때 실행
error Function
(jqXHR jqXHR, String textStatus, String errorThrown) 		요청작업이 실패했을때 실행
complete Function
(jqXHR jqXHR, String textStatus) 		요청작업이 종료 되었을때 실행
