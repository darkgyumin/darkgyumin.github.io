AngularJS

Angular-cli 명령어

//angular-cli install
npm install -g @angular/cli

//angular-cli update
ng update -g @angular/cli

//angular-cli delete
ng uninstall -g @angular/cli

//project create
ng new toh

//development server execute
ng serve --port 4200 --open

//component create
ng g component heroes

//bootstrap, ng-bootstrap install
npm install --save bootstrap
npm install --save @ng-bootstrap/ng-bootstrap

//component create
ng g component 컴포넌트명

//service create
ng g service 서비스명
