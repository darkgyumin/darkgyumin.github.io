부모 컴포넌트에서 자식 컴포넌트로 데이터를 전달한다.
<컴포넌트명 속성명="속성값">Children</컴포넌트명>의 형태로 하위에 데이터를 전달할 수 있다.
자식 컴포넌트에서는 속성값은 this.props.속성명, Children값은 this.props.children의 형태로 전달받아 사용할 수 있다.
App.js

import React, { Component } from 'react';
import ListItem from './components/ListItem';

class App extends Component {
  render() {
    return (
      <div>
        <ListItem age="36">GyuMin</ListItem>
      </div>
    );
  }
}

export default App;
ListItem.js

import React, { Component } from 'react';

class ListItem extends Component {
    render() {
        return (
            <div>
                {this.props.age} {this.props.children}             
            </div>
        );
    }
}

export default ListItem;
