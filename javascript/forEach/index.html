var array = [1, 2, 3, 4, 5];

array.forEach(function(currentValue, index, array) {

});
map
  • 각각의 배열 값에 제공된 메소드를 실행한 후 값을 return한다.

var array = [1, 2, 3, 4, 5];

array.map(function(currentValue, index, array) {

});
filter
  • 각각의 배열 값에 제공된 메소드가 true일 경우만 값을 return한다.

var array = [1, 2, 3, 4, 5];

array.filter(function(condition, index, array) {

});
some
  • 각각의 배열 값이 조건을 일부만 만족하면 true를 return한다.

var array = [1, 2, 3, 4, 5];

array.some(function(condition) {

});
every
  • 각각의 배열 값에 조건을 모두 만족하면 true를 return한다.

var array = [1, 2, 3, 4, 5];

array.every(function(condition) {

});
