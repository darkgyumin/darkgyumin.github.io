Template literal
멀티라인 문자열과 보간기능을 사용하며 백틱(``)을 사용한다. 

let string = `string
text`;

console.log(string);
//string
//text


let val = '김규민';
let string = `string text ${val}`;

console.log(string);
//string text 김규민

let price = 1000;
let num = 5;

let string = `this price is ${price * num}`

console.log(string);
//this price is 5000
