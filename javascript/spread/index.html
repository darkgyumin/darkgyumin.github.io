Spread 연산자
(...)로 작성하며 배열을 다른 배열의 내부에 삽입하고 다양한 형태로 사용이 가능하다.
배열 결합

//ES5
var a=[1, 2, 3], b='val', c=[4, 'val2'], d;
d = a.concat(b, c);
console.log(d);
//[1, 2, 3, "val", 4, "val2"]

//ES6
let a=[1, 2, 3], b='val', c=[4, 'val2'], d;
d = [...a, b, ...c];

console.log(d);
//[1, 2, 3, "val", 4, "val2"]
객체 결합

let person = {name: 'GyuMin', gender: 'M'}
let animal = {type: 'Dog', age: 5}

let sum = {...person, ...animal}
중복제거

//Set : 중복이 불가능한 데이터 집합
let arr = [1, 2, 3, 4];
let arr2 = [4, 3, 5, 6];

let arrUnique = new Set([...arr, ...arr2]);
console.log(arrUnique);
//Set(6) {1, 2, 3, 4, 5, 6}
Apply 대체
[Function.prototype.apply]
myFunction실행시 배열값을 파라미터로 전달하여 각각의 배열값이 인자로 적용된다. 

function myFunction(x, y, z) {
    console.log(x, y, z);
}
var args = [0, 1, 2];

myFunction.apply(null, args);
Deep Copy

//Shallow Copy
let arr = [1, 2, 3];
let arr2 = arr;

arr2[0] = 200;

console.log(arr, arr2);
//(3) [200, 2, 3] (3) [200, 2, 3]

//Deep Copy
let arr = [1, 2, 3];
let arr2 = [...arr];

arr2[0] = 200;

console.log(arr, arr2);
//(3) [1, 2, 3] (3) [200, 2, 3]
