Array
  • 공통된 관심사를 가진 값의 묶음
  • 자바스크립트는 배열에 다양한 형태의 값을 저장할 수 있다.
배열객체 생성방법 2가지

//리터럴 방식
var arrLiteral1 = [];	
var arrLiteral2 = [,,,,,];	//크기가 5인 배열을 생성
var arrLiteral3 = ['1', true, 33];	//지정한 값으로 배열을 생성

//생성자 방식
var arrConstructor1 = new Array();	
var arrConstructor2 = new Array(5);	//크기가 5인 배열을 생성
var arrConstructor3 = new Array('1', true, 33);	//파라미터로 넘어온 데이터로 배열을 생성
배열 값 추가, 제거
Keyword Description
push 배열의 끝에 새로운 값을 추가한다.
unshift 배열의 앞에 새로운 값을 추가한다.
pop 배열의 끝의 값을 삭제한다.
shift 배열의 앞의 값을 삭제한다. 

//push
var arr = [];
arr.push('jsp');
arr.push('php');
arr.push('asp');

console.log(arr);	//(3) ["jsp", "php", "asp"]

//pop
arr.pop();
console.log(arr);	//(3) (2) ["jsp", "php"]

//unshift
var arr = [];
arr.unshift('jsp');
arr.unshift('php');
arr.unshift('asp');

console.log(arr);	//(3) ["asp", "php", "jsp"]

//shift
arr.shift();
console.log(arr);	//(3) ["php", "jsp"]
