Scope
  • 선언된 변수가 사용가능한 유효범위를 나타낸다.
전역변수와 지역변수

//전역변수
var x = 'global';	

function fnExec() {
	//지역변수
	var x = 'local';
	x = 'change';
}
fnExec();
console.log(x);	//global
렉시컬스코핑(Lexical Scoping, Static Scoping)
  • 함수 실행시 유효범위를 실행환경이 아닌 함수정의환경을 기준으로 참조하는 특성이다.

var val = 'global';
function fnInner() {
	console.log(val);	//global
}

function fnOuter() {
	var val = 'logical';
  fnInner();
}

fnOuter();
