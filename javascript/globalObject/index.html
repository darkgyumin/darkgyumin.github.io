Global Object
  • 모든 객체의 최상위 객체를 의미하며 모든 객체는 전역객체의 프로퍼티와 메소드이다.
  • Browser-side에서는 window, Server-side에서는 global을 의미한다.
  • 브라우저에서 객체를 명시하지 않으면 암묵적으로 window의 프로퍼티로 간주한다.

var varVal = 'varValue';

console.log(varVal);
console.log(window.varVal);

//출력 varValue
//출력 varValue

function fnSample() {
	console.log('fnSample');
}

fnSample();
window.fnSample();

//출력 fnSample
//출력 fnSample
